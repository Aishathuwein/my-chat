rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow all users to read users collection
    match /users/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Messages rules
    match /messages/{messageId} {
      // Allow reading if user is authenticated
      allow read: if request.auth != null;
      
      // Allow writing if user is authenticated
      allow create: if request.auth != null;
      
      // Allow updating/deleting only for sender or admin
      allow update, delete: if request.auth != null && 
        (request.auth.uid == resource.data.senderId || 
         isAdmin(request.auth.uid, resource.data.chatId));
    }
    
    // Chats rules
    match /chats/{chatId} {
      // Allow reading if user is a participant
      allow read: if request.auth != null && 
        request.auth.uid in resource.data.participants;
      
      // Allow writing if user is a participant
      allow write: if request.auth != null && 
        request.auth.uid in resource.data.participants;
    }
    
    // Groups rules
    match /groups/{groupId} {
      // Allow reading if user is a member
      allow read: if request.auth != null && 
        request.auth.uid in resource.data.members;
      
      // Allow writing if user is creator or admin
      allow write: if request.auth != null && 
        (request.auth.uid == resource.data.createdBy || 
         request.auth.uid in resource.data.admins);
    }
    
    // Typing indicators
    match /typing/{docId} {
      allow read, write: if request.auth != null;
    }
    
    // Helper function to check if user is admin
    function isAdmin(userId, chatId) {
      let groupDoc = get(/databases/$(database)/documents/groups/$(chatId));
      return groupDoc != null && userId in groupDoc.data.admins;
    }
  }
}
